```python
from abc import ABC, abstractmethod
from typing import List, Dict, Any, Optional

class BaseAbility(ABC):
    """Classe de base pour toutes les capacités individuelles"""
    
    def __init__(self, hero_code: str, ability_number: int, name: str, description: str):
        self.hero_code = hero_code
        self.ability_number = ability_number  
        self.name = name
        self.description = description
    
    @abstractmethod
    def execute(self, caster, targets: List, context: Dict[str, Any], log: List[str]) -> bool:
        """Exécute la capacité avec effets mécaniques réels"""
        pass
    
    def get_preview(self) -> str:
        """Retourne un aperçu des effets de la capacité"""
        return f"📋 {self.name}: Effet mécanique défini"
```

3. **Créer `ability_registry.py`** (45 min)
```python
from typing import Dict, Type
from .base_ability import BaseAbility

class AbilityRegistry:
    """Registre central des 48 capacités individuelles"""
    
    def __init__(self):
        self._abilities: Dict[str, Type[BaseAbility]] = {}
    
    def register(self, ability_class: Type[BaseAbility]):
        """Enregistre une capacité dans le registre"""
        key = f"{ability_class.hero_code}_{ability_class.ability_number}"
        self._abilities[key] = ability_class
    
    def get_ability(self, hero_code: str, ability_number: int) -> Optional[Type[BaseAbility]]:
        """Récupère une capacité par son code héros et numéro"""
        key = f"{hero_code}_{ability_number}"
        return self._abilities.get(key)

# Instance globale du registre
ABILITY_REGISTRY = AbilityRegistry()
```
